let usuario = {
    nombre: 'David',
    apellido: 'Thoreau',
    tercerCampo: 'Walden'
}

//Más facil pordemos ver como creamos una variable apellido con el mismo nombre que la propiedad.
//tambien podemos poner un valor por defecto por si no encontramos la propiedad en el objeto.

var { nombre: alias, apellido, edad = 0 } = usuario;

console.log('variable apellido obtenida a partir de la propiedad apellido ->', apellido);
console.log('variable edad cullo valor por defecto es 0 -> ', edad);
